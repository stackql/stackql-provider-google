```sql
name,

displayName,
hostType,

satisfiesPzs,
state,
type,
uid,
updateTime,
uri

WHERE projectsId = '{{ projectsId }}' -- required
AND locationsId = '{{ locationsId }}' -- required
AND appGatewaysId = '{{ appGatewaysId }}' -- required;
```
